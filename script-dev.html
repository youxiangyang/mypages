开发环境配置

windows激活虚拟环境

@echo off
REM 设置虚拟环境的路径
set VENV_PATH=C:\path\to\your\venv

REM 检查虚拟环境目录是否存在
if not exist "%VENV_PATH%" (
    echo 虚拟环境目录不存在: %VENV_PATH%
    exit /b 1
)

REM 激活虚拟环境
call "%VENV_PATH%\Scripts\activate.bat"

REM 提示用户虚拟环境已激活
echo 虚拟环境已激活: %VENV_PATH%

REM 保持命令提示符窗口打开
cmd /k

MySQL 初始化和启动脚本

配置文件样例:

[mysql]
# 设置mysql客户端默认字符集
default-character-set=utf8mb4
[mysqld]
# 设置3306端口
port=3306
# 设置mysql的安装目录
basedir=D:\\soft\\database\\MySQL\\mysql-8.0.25-winx64
datadir=D:\\soft\\database\\MySQL\\mysql-8.0.25-winx64\\data
# 允许最大连接数
max_connections=20
# 服务端使用的字符集默认为8比特编码的latin1字符集
character-set-server=utf8mb4
# 创建新表时将使用的默认存储引擎
default-storage-engine=INNODB

命令行:

.\mysqld.exe --defaults-file=D:\soft\database\MySQL\mysql-8.0.25-winx64\init.ini --initialize --console

不使用配置文件:

.\mysqld.exe --initialize --console

修改密码:

ALTER USER 'root'@'localhost' IDENTIFIED BY '新密码';

启动脚本:

@echo off
@echo Mysql startup ...
start bin\mysqld.exe 
exit

Minio 启动脚本

REM 设置MinIO可执行文件的路径
set MINIO_EXE=C:\soft\minio\minio.exe

REM 设置数据存储路径
set DATA_DIR=C:\soft\minio\data

REM 设置访问密钥和秘密密钥
set ACCESS_KEY=PoKA5wjIviAbqIbnElWY
set SECRET_KEY=bSHt5Yd3wE3VV2CIWHLf9FrwSsH6FJDoIvBLYLJE

REM 设置端口号
set PORT=9000

REM 启动MinIO服务
%MINIO_EXE% server %DATA_DIR% --address ":%PORT%" --console-address ":9001"

pause
